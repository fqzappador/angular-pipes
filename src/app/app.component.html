Angular Pipes


Pipes allow us to visually transform our data without changing its real value, in this way we can show data with a better user experience, below are the most used examples:
Variable Pipe ( | ) Output
{{ name }} uppercase {{ name | uppercase }}
{{ name }} lowercase {{ name | lowercase }}
{{ name }} slice:3 {{ name | slice:3 }}
{{ name }} slice:0:3 {{ name | slice:0:3 }}
{{ array }} slice:1:5 {{ array | slice:1:5 }}
{{ PI }} number:'3.0-0' (decimal) {{ PI | number:'3.0-0' }}
{{ PI }} number:'.0-2' (decimal) {{ PI | number:'.0-2' }}
{{ percentages }} percent {{ percentages | percent }}
{{ percentages }} percent:'2.2-2' {{ percentages | percent:'2.2-2' }}
{{ salary }} currency (USD for default) {{ salary | currency }}
{{ salary }} currency:'EUR' {{ salary | currency:'EUR' }}
{{ salary }} currency:'GBP' {{ salary | currency:'GBP' }}
{{ salary }} currency:'CAD':'symbol-narrow':'.0-0' (without decimal) {{ salary | currency:'CAD':'symbol-narrow':'.0-0' }}
{{ promiseValue }} async (with setTimeout) {{ promiseValue | async }}
{{ date }} date {{ date | date }}
{{ date }} date:'medium' {{ date | date:'medium' }}
{{ date }} date:'short' {{ date | date:'short' }}
{{ date }} date:'MMMM - dd' {{ date | date:'MMMM - dd' }}
{{ date }} date:'MMMM - dd':'':'fr' {{ date | date:'MMMM - dd':'':'fr' }}
{{ date }} date:'MMMM - dd':'':'es' {{ date | date:'MMMM - dd':'':'es' }}
{{ date }} date:'MMMM - dd':'':'{{ language }}' (click event) (custom pipe)
{{ date | date:'MMMM - dd':'':language }}
{{ name2 }} capitalized (custom pipe) {{ name2 | capitalized:true }}
{{ name }} password:{{ enable }} (click event) (custom pipe)
{{ name | password:enable }}
